/* 

## PM Explanation: ##

This code is registering two actions, 
one when the painting's pointer enters and another when it leaves. 
The first action scales the painting to twice its size on hover, 
while the second scales it back down to 1x1 on hover.

## Dev Explanation: ##

--Variables:--
painting: the canvas
scene: the scene
painting.actionManager: a BABYLON.ActionManager object that is used
  to register actions for the canvas
painting.scaling: a new BABYLON.Vector3 object that is set to 2, 2, 2 
  when OnPointerOverTrigger is called
  and 1, 1, 1 when OnPointerOutTrigger is called

--What's it Doing:--
The code is registering two actions,
one for when the painting pointer enters 
and another for when it leaves.
The first action will scale the painting to a 2x2 square on hover.
The second action scales the painting back down to 1x1 on hover

*/
var scaleOnHover = (painting, scene) => {
    painting.actionManager = new BABYLON.ActionManager(scene);

    painting.actionManager.registerAction(
        new BABYLON.ExecuteCodeAction({
            trigger: BABYLON.ActionManager.OnPointerOverTrigger,
        }, function () {
            painting.scaling = new BABYLON.Vector3(2, 2, 2);
        })
    );

    painting.actionManager.registerAction(
        new BABYLON.ExecuteCodeAction({
            trigger: BABYLON.ActionManager.OnPointerOutTrigger,
        }, function () {
            painting.scaling = new BABYLON.Vector3(1, 1, 1);
        })
    );
}
